    /* ===== GERADOR ===== */
    function gerarJogo() {
      const pares = [];
      const impares = [];

      for (let i = 1; i <= 25; i++) {
        (i % 2 === 0 ? pares : impares).push(i);
      }

      const jogo = new Set();

      while (jogo.size < 7) {
        jogo.add(pares[Math.floor(Math.random() * pares.length)]);
      }

      while (jogo.size < 15) {
        jogo.add(impares[Math.floor(Math.random() * impares.length)]);
      }

      const resultado = Array.from(jogo).sort((a, b) => a - b);

      document.getElementById("resultado").innerText =
        resultado.map(n => n.toString().padStart(2, "0")).join(" ‚Ä¢ ");

      mostrarEstatisticas(resultado);
      salvarHistorico(resultado);
      carregarHistorico();
    }

    /* ===== ESTAT√çSTICAS ===== */
    function mostrarEstatisticas(jogo) {
      const pares = jogo.filter(n => n % 2 === 0).length;
      const soma = jogo.reduce((a, b) => a + b, 0);

      document.getElementById("estatisticas").innerHTML =
        `üìä Pares: ${pares} | √çmpares: ${15 - pares}<br>üî¢ Soma: ${soma}`;
    }

    /* ===== HIST√ìRICO ===== */
    function salvarHistorico(jogo) {
      let h = JSON.parse(localStorage.getItem("historico")) || [];
      h.unshift({ data: new Date().toLocaleString(), jogo });
      localStorage.setItem("historico", JSON.stringify(h.slice(0, 10)));
    }

    function carregarHistorico() {
      let h = JSON.parse(localStorage.getItem("historico")) || [];
      let html = "<strong>üìÅ Hist√≥rico</strong><br>";

      });

    }
